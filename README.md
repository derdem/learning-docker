docker run -p 3306:3306 --name database-mysql -e MYSQL_ROOT_PASSWORD=envelio123 -d mysql:latest

## enter container

docker compose exec database mysql -u root -p

SHOW DATABASES;

USE app;

## fill database with data from sql file

source application/setup.sql

SHOW TABLES;
